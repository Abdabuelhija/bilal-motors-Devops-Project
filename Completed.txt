1.build image
docker build -t bilal-motors-frontend:1.0 .
docker build -t bilal-motors-backend:1.0 .
2. Push an Image to Docker Hub:

docker tag bilal-motors-frontend:1.0 abedabu28/bilal-motors-frontend:1.0
docker push abedabu28/bilal-motors-frontend:1.0

docker tag bilal-motors-backend:1.0 abedabu28/bilal-motors-backend:1.0
docker push abedabu28/bilal-motors-backend:1.0

3.helm create bilal-motors-backend
helm create bilal-motors-frontend

4- run :
helm upgrade -i backend-release "charts/backend-chart" --values "charts/backend-chart/values.yaml"
helm upgrade -i frontend-release "charts/frontend-chart" --values "charts/frontend-chart/values.yaml"

-kubectl -n default port-forward svc/backend-release-backend-chart 8000:8000
- kubectl -n default port-forward svc/frontend-release-frontend 3000:80
